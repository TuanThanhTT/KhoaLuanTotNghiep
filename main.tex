% main.tex – PHIÊN BẢN HOÀN CHỈNH CUỐI CÙNG (đúng chuẩn ĐH Việt Nam 2025)
% Số trang: MỤC LỤC = i, luôn ở giữa phía dưới
% Biên dịch: XeLaTeX → Biber → XeLaTeX → XeLaTeX
% =============================================================================
\documentclass[12pt,a4paper,oneside,openany]{book}

% ==================== FONT & TIẾNG VIỆT ====================
\usepackage{fontspec}
\setmainfont{Times New Roman}
\usepackage[vietnamese]{babel}

% ==================== LỀ TRANG & DÒNG ====================
\usepackage{geometry}
\geometry{
  a4paper,
  top=3cm,
  bottom=2.5cm,
  left=3cm,
  right=2cm,
}

\usepackage{setspace}
\onehalfspacing
\setlength{\parindent}{1.25cm}

% ==================== SỐ TRANG LUÔN Ở DƯỚI GIỮA ====================
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}                        % xóa hết header/footer cũ
\fancyfoot[C]{\thepage}           % số trang ở giữa chân trang
\renewcommand{\headrulewidth}{0pt}
\setlength{\headheight}{15pt}

% Trang mở đầu chương (plain) cũng phải ở dưới
\fancypagestyle{plain}{
  \fancyhf{}
  \fancyfoot[C]{\thepage}
  \renewcommand{\headrulewidth}{0pt}
}

% ==================== ĐỊNH DẠNG CHƯƠNG ====================
\usepackage{titlesec}
\titleformat{\chapter}[display]
  {\centering\normalfont\fontsize{14}{18}\selectfont\bfseries}
  {CHƯƠNG \thechapter}{10pt}{\LARGE\bfseries}
\titlespacing*{\chapter}{0pt}{30pt}{20pt}

% Chương đặc biệt không số (MỞ ĐẦU, KẾT LUẬN, TÀI LIỆU THAM KHẢO...)
\newcommand{\specialchapter}[1]{%
  \cleardoublepage
  \thispagestyle{empty}
  \vspace*{5cm}
  {\centering\fontsize{16}{20}\selectfont\bfseries #1\par}
  \vspace{6cm}
  \addcontentsline{toc}{chapter}{#1}%
  \cleardoublepage
  \pagestyle{fancy}
}

% ==================== TOÁN HỌC, BẢNG, ẢNH ====================
\usepackage{amsmath,amssymb,amsthm}
\numberwithin{equation}{chapter}
\numberwithin{table}{chapter}
\numberwithin{figure}{chapter}

\usepackage{booktabs,tabularx,caption,float}
\captionsetup[table]{labelfont=bf,labelsep=quad,name=Bảng,justification=centering}
\captionsetup[figure]{labelfont=bf,labelsep=quad,name=Ảnh,position=bottom}

% ==================== KHÁC ====================
\usepackage{tcolorbox}
\tcbuselibrary{skins}

\usepackage[unicode,hidelinks]{hyperref}
\usepackage[capitalize]{cleveref}

\renewcommand{\listtablename}{Danh mục bảng}
\renewcommand{\listfigurename}{Danh mục ảnh}

% Đánh số phụ lục PL1, PL2...
\newcommand{\appendixpagenumber}{%
  \renewcommand{\thepage}{PL\arabic{page}}%
}

% =============================================================================
\begin{document}

% ==================== TRANG BÌA (không số trang) ====================
\include{frontmatter/cover}
\include{frontmatter/subcover}

% ==================== CAM KẾT & LỜI CẢM ƠN (không số trang) ====================
\pagestyle{empty}
\include{frontmatter/declaration}
\cleardoublepage
\include{frontmatter/acknowledgement}
\cleardoublepage

% ==================== BẮT ĐẦU TỪ ĐÂY LÀ TRANG i ====================
\pagestyle{fancy}           % bật số trang ở dưới
\pagenumbering{roman}       % số La Mã
\setcounter{page}{1}        % MỤC LỤC = trang i

\tableofcontents
\cleardoublepage

\listoftables
\cleardoublepage

\listoffigures
\cleardoublepage

% ==================== NỘI DUNG CHÍNH (từ trang 1) ====================
\mainmatter
\pagenumbering{arabic}

\include{chapters/introduction}
\include{chapters/chapter1}
\include{chapters/chapter2}
\include{chapters/chapter3}
\include{chapters/chapter4}
\include{chapters/conclusion}

% ==================== TÀI LIỆU THAM KHẢO ====================
\cleardoublepage
\bibliographystyle{ieeetr}
\bibliography{bib/refs}

% ==================== PHỤ LỤC ====================
\appendix
\cleardoublepage
\appendixpagenumber        % đổi thành PL1, PL2...
\pagestyle{fancy}          % vẫn giữ số trang ở dưới
\setcounter{page}{1}       % phụ lục bắt đầu từ PL1

\include{appendices/appendixA}
\include{appendices/appendixB}
% Thêm phụ lục khác nếu cần

\end{document}
